
/*Total entity*/
With S1 as 
(
	 select a, b, c, d
	 from
	 Table1
	 where 'BI_100' in (ColA,ColB,ColC)
),
S2 as 
(
	 select a, b, c * -1 as c, d * -1 as d
	 from
	 Table1
	 where 'BI_200' in (business_code)
)
select a, b, sum(c) as c, sum(d) as d
from
(
	select * from S1 union all select * from S2
)
group by 1,2,3,4;
